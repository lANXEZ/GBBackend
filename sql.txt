-- Create and select the database
DROP DATABASE IF EXISTS gymbro;
CREATE DATABASE IF NOT EXISTS gymbro;
USE gymbro;

-- User Table
CREATE TABLE User ( 
    UserID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    Username VARCHAR(50) NOT NULL, 
    Password VARCHAR(255) NOT NULL, 
    FirstName VARCHAR(20) NOT NULL, 
    LastName VARCHAR(20) NOT NULL, 
    DoB DATE NOT NULL, 
    Status VARCHAR(20) NOT NULL 
);

-- WorkoutPlan Table
CREATE TABLE WorkoutPlan ( 
    PlanID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    PlanName VARCHAR(255) NOT NULL,
    UserID BIGINT NOT NULL, 
    Type CHAR(1) NOT NULL, 
    ProviderID BIGINT,
    FOREIGN KEY (UserID) REFERENCES User(UserID),
    FOREIGN KEY (ProviderID) REFERENCES User(UserID)
);

-- WorkingDay Table
CREATE TABLE WorkingDay ( 
    WorkingDayID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    Day INT NOT NULL 
);

-- ExerciseMoves Table
CREATE TABLE ExerciseMoves ( 
    ExMoveID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    Steps TEXT NOT NULL, 
    Description TEXT NOT NULL, 
    Caution TEXT, 
    URL VARCHAR(500), 
    Accessibility VARCHAR(10) NOT NULL, 
    UserID BIGINT, 
    RecordType VARCHAR(20) NOT NULL, 
    ProgressType VARCHAR(10) NOT NULL, 
    FOREIGN KEY (UserID) REFERENCES User(UserID)
);

-- WorkoutRoutine Table (Joining Table - No Auto Increment needed for composite PK)
CREATE TABLE WorkoutRoutine ( 
    PlanID BIGINT NOT NULL, 
    WorkingDayID BIGINT NOT NULL, 
    PRIMARY KEY (PlanID, WorkingDayID),
    FOREIGN KEY (PlanID) REFERENCES WorkoutPlan(PlanID) ON DELETE CASCADE,
    FOREIGN KEY (WorkingDayID) REFERENCES WorkingDay(WorkingDayID) ON DELETE CASCADE
);

-- ExerciseList Table (Joining Table - No Auto Increment needed for composite PK)
CREATE TABLE ExerciseList ( 
    WorkingDayID BIGINT NOT NULL, 
    ExMoveID BIGINT NOT NULL, 
    PRIMARY KEY (WorkingDayID, ExMoveID),
    FOREIGN KEY (WorkingDayID) REFERENCES WorkingDay(WorkingDayID) ON DELETE CASCADE,
    FOREIGN KEY (ExMoveID) REFERENCES ExerciseMoves(ExMoveID) ON DELETE CASCADE
);

-- Session Table
CREATE TABLE Session ( 
    SessionID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    UserID BIGINT NOT NULL, 
    WorkingDayID BIGINT NOT NULL, 
    ExMoveID BIGINT NOT NULL,
    SessionDate DATE NOT NULL,
    FOREIGN KEY (UserID) REFERENCES User(UserID),
    FOREIGN KEY (WorkingDayID) REFERENCES ExerciseList(WorkingDayID),
    FOREIGN KEY (ExMoveID) REFERENCES ExerciseList(ExMoveID)
);

-- PersonalRecord Table
-- Note: Weight is DECIMAL(5,2) which allows max 999.99kg.
CREATE TABLE PersonalRecord ( 
    PRID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
    SessionID BIGINT NOT NULL, 
    Weight DECIMAL(5,2), 
    Rep INT, 
    Time INT, 
    FOREIGN KEY (SessionID) REFERENCES Session(SessionID) ON DELETE CASCADE
);

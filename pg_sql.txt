-- Reset the database by dropping existing tables
-- This avoids the "DROP DATABASE cannot run inside a transaction block" error
DROP TABLE IF EXISTS PersonalRecord CASCADE;
DROP TABLE IF EXISTS Session CASCADE;
DROP TABLE IF EXISTS ExerciseList CASCADE;
DROP TABLE IF EXISTS WorkoutRoutine CASCADE;
DROP TABLE IF EXISTS ExerciseMoves CASCADE;
DROP TABLE IF EXISTS WorkingDay CASCADE;
DROP TABLE IF EXISTS WorkoutPlan CASCADE;
DROP TABLE IF EXISTS TrainingList CASCADE;
DROP TABLE IF EXISTS "User" CASCADE;

-- User Table
CREATE TABLE "User" ( 
    UserID BIGSERIAL PRIMARY KEY, 
    Username VARCHAR(50) NOT NULL, 
    Password VARCHAR(255) NOT NULL, 
    FirstName VARCHAR(20) NOT NULL, 
    LastName VARCHAR(20) NOT NULL, 
    DoB DATE NOT NULL, 
    Status VARCHAR(20) NOT NULL 
);

-- TrainingList Table (NEW: Maps Trainers to Clients)
CREATE TABLE TrainingList (
    TrainerID BIGINT NOT NULL,
    ClientID BIGINT NOT NULL,
    PRIMARY KEY (TrainerID, ClientID),
    FOREIGN KEY (TrainerID) REFERENCES "User"(UserID) ON DELETE CASCADE,
    FOREIGN KEY (ClientID) REFERENCES "User"(UserID) ON DELETE CASCADE
);

-- WorkoutPlan Table
CREATE TABLE WorkoutPlan ( 
    PlanID BIGSERIAL PRIMARY KEY, 
    PlanName VARCHAR(255) NOT NULL,
    UserID BIGINT NOT NULL, 
    Type CHAR(1) NOT NULL, 
    ProviderID BIGINT,
    FOREIGN KEY (UserID) REFERENCES "User"(UserID),
    FOREIGN KEY (ProviderID) REFERENCES "User"(UserID)
);

-- WorkingDay Table
CREATE TABLE WorkingDay ( 
    WorkingDayID BIGSERIAL PRIMARY KEY, 
    Day INT NOT NULL 
);

-- ExerciseMoves Table
CREATE TABLE ExerciseMoves ( 
    ExMoveID BIGSERIAL PRIMARY KEY, 
    Steps TEXT NOT NULL, 
    Description TEXT NOT NULL, 
    Caution TEXT, 
    URL VARCHAR(500), 
    Accessibility VARCHAR(10) NOT NULL, 
    UserID BIGINT, 
    RecordType VARCHAR(20) NOT NULL, 
    ProgressType VARCHAR(10) NOT NULL, 
    FOREIGN KEY (UserID) REFERENCES "User"(UserID)
);

-- WorkoutRoutine Table (Joining Table - No Auto Increment needed for composite PK)
CREATE TABLE WorkoutRoutine ( 
    PlanID BIGINT NOT NULL, 
    WorkingDayID BIGINT NOT NULL, 
    PRIMARY KEY (PlanID, WorkingDayID),
    FOREIGN KEY (PlanID) REFERENCES WorkoutPlan(PlanID) ON DELETE CASCADE,
    FOREIGN KEY (WorkingDayID) REFERENCES WorkingDay(WorkingDayID) ON DELETE CASCADE
);

-- ExerciseList Table (Joining Table - No Auto Increment needed for composite PK)
CREATE TABLE ExerciseList ( 
    WorkingDayID BIGINT NOT NULL, 
    ExMoveID BIGINT NOT NULL, 
    PRIMARY KEY (WorkingDayID, ExMoveID),
    FOREIGN KEY (WorkingDayID) REFERENCES WorkingDay(WorkingDayID) ON DELETE CASCADE,
    FOREIGN KEY (ExMoveID) REFERENCES ExerciseMoves(ExMoveID) ON DELETE CASCADE
);

-- Session Table (UPDATED: Renamed to UserWeight and UserHeight)
CREATE TABLE Session ( 
    SessionID BIGSERIAL PRIMARY KEY, 
    UserID BIGINT NOT NULL, 
    WorkingDayID BIGINT NOT NULL, 
    ExMoveID BIGINT NOT NULL,
    SessionDate DATE NOT NULL,
    UserWeight DECIMAL(5,2),
    UserHeight DECIMAL(5,2),
    FOREIGN KEY (UserID) REFERENCES "User"(UserID) ON DELETE CASCADE,
    FOREIGN KEY (WorkingDayID, ExMoveID) REFERENCES ExerciseList(WorkingDayID, ExMoveID) ON DELETE CASCADE
);

-- PersonalRecord Table
-- Note: Weight is DECIMAL(5,2) which allows max 999.99kg.
CREATE TABLE PersonalRecord ( 
    PRID BIGSERIAL PRIMARY KEY, 
    SessionID BIGINT NOT NULL, 
    Weight DECIMAL(5,2), 
    Rep INT, 
    Time INT, 
    FOREIGN KEY (SessionID) REFERENCES Session(SessionID) ON DELETE CASCADE
);
